$$
\lim_{n \rightarrow \infty}{\frac{\sqrt[3]{n^2}\sin{n!}}{n+1}}
$$

\medskip

$$
\lim_{n \rightarrow \infty}{\frac{\sqrt[3]{n^2}\sin{n!}}{n+1}} = \lim_{n \rightarrow \infty}{\frac{\sqrt[3]{n^2}\sin{n!} : n^{\frac{2}{3}}}{(n+1) : n^{\frac{2}{3}}}} = \lim_{n \rightarrow \infty}{\frac{\sin{n!}}{n^{\frac{1}{3}}+\frac{1}{n^{2/3}}}}
$$
$$
-1 \leq \sin{n!} \leq 1
$$
$$
\left|{\frac{\sin{n!}}{n^{\frac{1}{3}} + 1}}\right| \leq {\frac{1}{n^{\frac{1}{3}} + 1}} \leq {\frac{1}{n^{\frac{1}{3}} }} \rightarrow0
$$
