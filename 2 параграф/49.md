$$
\lim_{n \rightarrow \infty}{\frac{(-2)^n + 3^n}{(-2)^{n+1} + 3^{n+1}}}
$$

\medskip

$$
\lim_{n \rightarrow \infty}{\frac{(-2)^n + 3^n}{(-2)^{n+1} + 3^{n+1}}} = \lim_{n \rightarrow \infty}{\frac{((-2)^n + 3^n):3^n}{((-2)^{n+1} + 3^{n+1}):3^n}} =
$$
$$
= \lim_{n \rightarrow \infty}{\frac{\left(\frac{-2}{3}\right)^n + 1}{-2 \cdot \left(\frac{-2}{3}\right)^{n} + 3}} = \frac{1}{3}
$$
Так как $\left|-\frac{2}{3}\right| < 1$, то $\left( \frac{-2}{3} \right)^n \rightarrow 0$
